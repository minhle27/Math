\definecolor{claimblue}{HTML}{315E81}

% Syntax: \newtcbtheorem[init options]{EnvironmentName}{Display Name}{Box Options}{Reference Prefix}
% The 'O{} m m m' tells LaTeX to expect 1 Optional and 3 Mandatory arguments:
% {Title}{Label}{Source}
\NewTColorBox[use counter=problem]{problem}{ O{} m m m }{
    enhanced,
    sharp corners,
    frame hidden,
    boxrule=0pt,
    colback=claimbg,
    borderline west={3pt}{0pt}{claimblue},
    drop fuzzy shadow,
    coltitle=claimblue,
    fonttitle=\bfseries,
    % Formats the title as: Problem X (Title). 
    title={Problem \thetcbcounter\ifstrempty{#2}{}{ (#2)}.\ },
    label={prob:#3}, % Automatically prefixes your label with 'prob:'
    attach title to upper={\par\vspace{1ex}}, % Keeps the title inline with the text
    % Places the source flush right at the bottom, without a dashed line
    after upper={\ifstrempty{#4}{}{\par\vspace{1ex}\noindent\hfill\footnotesize\textit{Source: #4}}},
    #1 % Allows you to pass optional tcolorbox settings if needed
}

\NewTColorBox{solution}{ O{} }{
    enhanced,
    breakable, 
    frame hidden,
    boxrule=0pt,
    colback=white, 
    borderline west={2pt}{0pt}{gray!40!white},

    fonttitle=\bfseries\itshape, 
    title={Solution:},

\definecolor{lemmagreen}{HTML}{2E7D32} % Deep, professional green
\definecolor{lemmabg}{HTML}{E8F5E9}    % Very soft green background

% --- The Lemma Environment ---
\NewTColorBox[use counter=lemma]{lemma}{ O{} m m }{
    enhanced,

    frame hidden,
    boxrule=0pt,
    colback=lemmabg,

    code={\ifstrempty{#3}{}{\tcbset{label={lem:#3}}}},
    attach title to upper={\par\vspace{1ex}},
    #1
}
