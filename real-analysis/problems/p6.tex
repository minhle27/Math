\begin{problem}{}{}{Real Mathematical Analysis, Pugh, Ex.78 p.131}
$(p_1,\dots,p_n)$ is an \textbf{$\epsilon$-chain} in a metric space $M$ if for each $i$ we have $p_i \in M$ and $d(p_i,p_{i+1}) < \epsilon$. 
The metric space is \textbf{chain-connected} if for each $\epsilon > 0$ and each pair of points $p,q \in M$ there is an $\epsilon$-chain from $p$ to $q$.
\begin{enumerate}
    \item[(a)] Show that every connected metric space is chain-connected.
    \item[(b)] Show that if $M$ is compact and chain-connected then it is connected.
    \item[(c)] Is $\mathbb{R} \setminus \mathbb{Z}$ chain-connected?
    \item[(d)] If $M$ is complete and chain-connected, is it connected?
\end{enumerate}
\end{problem}
